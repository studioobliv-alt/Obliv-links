<script>

  document.getElementById("botao-download").addEventListener("click", () => {

    const tamanho = 1024 * 1024 * 1024 * 2; // 2 GB

    const dados = new Uint8Array(tamanho); // Preenchido com zeros

    const blob = new Blob([dados], { type: "application/octet-stream" });

    const url = URL.createObjectURL(blob);

    

    const a = document.createElement("a");

    a.href = url;

    a.download = "arquivo-grande.zip";

    document.body.appendChild(a);

    a.click();

    document.body.removeChild(a);

  });

</script>
